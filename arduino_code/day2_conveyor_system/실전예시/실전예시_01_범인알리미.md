String analyzeColor(int r, int g, int b) {
  // ë°ê¸° ê³„ì‚°
  int brightness = (r + g + b) / 3;
  
  // ì–´ë‘ìš´ ìƒ‰ (ê²€ì€ìƒ‰)
  if (brightness < 50) {
    return "BLACK";
  }
  
  // ë°ì€ ìƒ‰ (í°ìƒ‰)
  if (brightness > 200 && abs(r - g) < 30 && abs(g - b) < 30) {
    return "WHITE";
  }
  
  // ìƒ‰ìƒ íŒë³„
  if (r > g && r > b) {
    if (g > 100) {
      return "YELLOW";  // ë¹¨ê°• + ì´ˆë¡ = ë…¸ë‘
    }
    return "RED";
  } else if (g > r && g > b) {
    return "GREEN";
  } else if (b > r && b > g) {
    return "BLUE";
  }
  
  return "UNKNOWN";
}
